  name: traefik
  namespace: traefik

  interval: 5m
  chart:
    spec:
      chart: traefik
      version: "24.0.0"
      sourceRef:
        kind: HelmRepository
        name: traefik
        namespace: flux-system

      retries: 3

      type: LoadBalancer

        exposedPort: 80

        port: 443
        expose: true
        exposedPort: 443
        protocol: TCP
    ingressRoute:
      dashboard:
        enabled: true

      kubernetesCRD:
        enabled: true
        allowCrossNamespace: true
        allowExternalNameServices: true

        enabled: true
        publishedService:
          enabled: true
